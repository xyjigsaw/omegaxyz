python代码实现: 
import math

def calEntropy(string):
    h = 0.0
    sumt = 0
    letter = [0] * 26
    string = string.lower()
    for i in range(len(string)):
        if string[i].isalpha():
            letter[ord(string[i]) - ord('a')] += 1
            sumt += 1
    print('\n', letter)
    for i in range(26):
        p = 1.0 * letter[i] / sumt
        if p > 0:
            h += -(p * math.log(p, 2))

    return h


test = input("输入一个英文句子：")
print('\n熵为：', calEntropy(test))
