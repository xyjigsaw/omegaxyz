Python密码强度判断

高级语言技术域PythonPython
不使用正则表达式，判断一个字符串是否满足强密码要求。（长度>=10，至少出现以下情况中的三种：大写字母、小写字母、数字、符号集（_！@#）:且不包括其它字符） 
def judge(pw):
    if len(pw) < 10:
        return False
    flag = [0, 0, 0, 0]
    symbol = ['_', '!', '@', '#']
    for i in pw:
        if i.isupper():
            flag[0] = 1
        elif i.islower():
            flag[1] = 1
        elif i.isdigit():
            flag[2] = 1
        elif i in symbol:
            flag[3] = 1
        else:
            return False
    if sum(flag) >= 3:
        return True
    else:
        return False

if __name__ == '__main__':
    while True:
        if judge(input('input:')):
            print('OK')
            break
        else:
            print('ERROR')
