Assembly language sorting algorithm

汇编语言Technology汇编语言

question

After using the variable to pass parameter keyboard input string into the buffer, sort the contents of the buffer in descending order and output (bubble sort)

process

①Input the string into the buffer through the keyboard ② Sort strings in descending order according to ASCII code (subroutine) ③Output the sorted strings (subroutine)

difficulty

Multiple loop implementation of sorting algorithm

code

;author XY

;数据区
Data segment
;注意此处一定要按照顺序写
buff db 16	;缓冲区大小
numb db '?' ;输入的数量自动统计
arry db 16 dup('?') ;缓冲区的内容一共16个字符
Data ends
;---------------------------

Code segment
assume cs:code,ds:data
main proc far
push ds
sub ax,ax
push ax
mov ds,ax
lea dx,buff
mov ah,10 ;即0A键盘输入至缓冲区
int 21h
call order
ret
main endp
;---------------------------

order proc near
mov cl,numb
mov ch,0
mov di,cx
lp1:mov cx,di	;外循环由di控制
	mov bx,0	;下标
lp2:mov al,arry[bx]	;内循环由cx控制
	cmp al,arry[bx+1]
	jge cont
	xchg al,arry[bx+1]
	mov arry[bx],al
cont:inc bx
	 loop lp2
	 dec di
	 jnz lp1
call output
ret
order endp
;---------------------------
output proc near
mov bl,numb
mov bh,0
mov byte ptr[arry+bx],'$'
mov dx,offset arry
mov ah,9
int 21h
ret
output endp
Code ends
end main

result
