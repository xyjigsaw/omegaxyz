C++STL implements depth and breadth first traversal of graphs (BFS&DFS)

Programming LanguagesC & C++Data StructuresTechnologyC++Data Structures
For basic operations and traversal of data structure graphs in C language (the storage structure is an adjacency matrix), please check:https://omegaxyz.com/2017/05/17/graphofds2/Please see the storage structure traversal of the adjacency list.https://omegaxyz.com/2017/05/16/graphofds/The following is the depth and breadth first traversal (BFS&DFS) of the C++STL implementation graph Among them, BFS needs to use a stack, and DFS needs to use a queue. The graph used in the algorithm below is:Code:
//Author: Xu Yi
//omegaxyz.com

#include<iostream>
#include<stack>
#include<queue>
#include<cstring>
#define MAX 100

using namespace std;

stack<int> BFS_Stack;
queue<int> DFS_Queue;
int G[MAX][MAX];
int visit[MAX];
int V,E;

void BFS(int s){
    cout<<"BFS"<<endl;
    int i,j,node;
    memset(visit,0,sizeof(visit));
    BFS_Stack.push(s);
    while(!BFS_Stack.empty()){
        node=BFS_Stack.top();
        BFS_Stack.pop();
        if(visit[node])continue;
        visit[node]=1;
        cout<<node<<"-->";
        for(i=0;i<V;i++){
            if(G[node][i])BFS_Stack.push(i);
        }
    }
    cout<<"NULL"<<endl;
}

void DFS(int s){
    cout<<"DFS"<<endl;
    int i,j,node;
    memset(visit,0,sizeof(visit));
    DFS_Queue.push(s);

    while(!DFS_Queue.empty()){
        node=DFS_Queue.front();
        DFS_Queue.pop();
        if(visit[node])continue;
        visit[node]=1;
        cout<<node<<"-->";
        for(i=0;i<V;i++)
            if(G[i][node])DFS_Queue.push(i);
        for(j=0;j<V;j++)
            if(G[node][j])DFS_Queue.push(j);
    }
    cout<<"NULL"<<endl;
}
