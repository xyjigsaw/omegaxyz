FIFO algorithm implements page fault scheduling for paging management

Programming LanguagesC & C++Technology操作系统C++操作系统原理
The basic principle of the FIFO (first in, first out) algorithm is: This algorithm always eliminates the page that enters the memory first, that is, it selects the page that has resided in the memory the longest and eliminates it.

data structure

First define a main memory tablememeoryListThe address space used to store the main memory is used for page conversion. Define anotherpageListas page table

replace

Note that here we first determine whether the current location is occupied, and then determine whether the page is at the current location. If it is necessary to interrupt the replacement, it is judged that because it is a FIFO algorithm, it is necessary to replace the one that has been stored for the longest time.The replaced position is updated according to the formula position = (position+1)%memoryList.size();

flow chart

shortcoming

The page fault rate of the FIFO algorithm accounts for 60%, so it is not suitable for use in modern operating systems.

code

#include<iostream>
#include<deque>
#include<cstdio>
#define NONE 999
using namespace std;

struct Lnode{
	int id;
};

deque<Lnode> memoryList;
deque<int> pageList;

void initMemory(int memorySize){
	for(int i=0;i<memorySize;i++){
		struct Lnode temp;
		temp.id = NONE;
		memoryList.push_back(temp);
	}

}

float FIFO(){
	int replace_count=0;
	int position=0;
	cout<<"\n页表      替换?         内容\n";
	for(int i=0;i<pageList.size();i++){
        cout<<pageList[i]<<"         ";
		int j=0;
		for(j=0;j<memoryList.size();j++){
			if(memoryList[j].id==NONE||memoryList[j].id==pageList[i]){
				memoryList[j].id=pageList[i];
				break;
			}
		}
		if(j==memoryList.size()){
			replace_count++;
			printf("%d被%d替换掉！  ", memoryList[position],pageList[i]);
			memoryList[position].id = pageList[i];
			position = (position+1)%memoryList.size();
		}else{
		    printf("              ");
		}
		for(j=0;j<memoryList.size();j++){
            if(memoryList[j].id!=NONE)
                cout<<memoryList[j].id<<" ";
            else
                cout<<" ";
		}
		cout<<endl;
	}
	return replace_count;
}


int main(){
	int num = 20;
	cout<<"输入流："<<endl;
	int arrayList[20] = {7,0,1,2,0,3,0,4,2,3,0,3,2,1,2,0,1,7,0,1};
	initMemory(3);
	for(int i=0;i<num;i++){
		pageList.push_back(arrayList[i]);
		cout<<pageList[i]<<" ";
	}
	cout<<endl;
	float replace_count = FIFO();

	cout<<"\n置换率为："<<replace_count<<"/"<<num<<"="<<replace_count/num<<endl;
}
