int LinkListInit(LinkList &L)
{
	LinkList tail, p;
	int n;
	p = (LNode*)malloc(sizeof(LNode));
	L = tail = p;
	L->next = NULL;
	printf("输入链表的初始长度:\n");
	scanf("%d", &n);
	L->data = n;
	printf("请输入%d个值:\n",n);
	while (n--)
	{
		p = (LNode*)malloc(sizeof(LNode));
		scanf("%d", &p->data);
		p->next = NULL;
		tail->next = p;
		tail = p;
	}
	p->next = NULL;
	printf("链表创建成功！\n");
	return OK;
}

③ Insert a value into the linked list
int LinkListInsert(LinkList &L, int i, ElemType e)
{
	LinkList p = L;
	LinkList s;
	int j = 0;
	while (p&&j < i - 1)
	{
		p = p->next; ++j;
	}
	if (!p || j>i - 1)return ERROR;
	s = (LinkList)malloc(sizeof(LNode));
	s->data = e; s->next = p->next; p->next = s;
	L->data++;
	return OK;
}
④ Delete a value in the linked list
int LinkListDelete(LinkList &L, int i, ElemType &e)
{
	LinkList p = L;
	LinkList q;
	int j = 0;
	while (p&&j < i - 1)
	{
		p = p->next; ++j;
	}
	if (!(p->next) || j>i - 1)return ERROR;
	q = p->next; p->next = q->next; e = q->data;
	free(q);
	L->data--;
	return OK;
}
⑤ Reverse of linked list
void LinkListReverse(LinkList &L)
{
	LinkList p, q, r;
	q = r = NULL;
	p=L->next;
	while (p->next)
	{
		q = p->next;
		p->next = r;
		r = p;
		p = q;
	}
	p->next = r; L->next->next = NULL;
	L->next = p;
}
Note: The insertion method used here has an inversion time efficiency of O(n) ⑥ Merge function of two non-decreasing linked lists
void LinkListMerge(LinkList &Lc, LinkList &La, LinkList &Lb)
{
	LinkList pa, pb, pc;
	pa = La->next;
	pb = Lb->next;
	Lc = pc = La;
	while (pa&&pb)
	{
		if (pa->data <= pb->data)
		{
			pc->next = (LNode*)malloc(sizeof(LNode));
			pc->next = pa; pc = pa; pa = pc->next;
		}
		else
		{
			pc->next = (LNode*)malloc(sizeof(LNode));
			pc->next = pb; pc = pb; pb = pb->next;
		}
	}
	pc = (LNode*)malloc(sizeof(LNode));
	pc->next = pa ? pa : pb;
	free(Lb);
}//非递减排列合并
⑦ Clearing and destroying linked lists Clear:
void ClearLinkList(LinkList &L){
	LinkList p, q;
	p = L->next;
	while (p)
	{
		q = p->next;
		free(p);
		p = q;
	}
	L->next = NULL;
}
destroy:
void DestroyLinklist(LinkList &L)
{
	LinkList p;
	while (L)
	{
		p = L->next;
		free(L);
		L = p;
	}
}

Summary 1. Pay attention to whether you want to take the lead node when defining the linked list. My linked list implements an optimized linked list of the lead node, that is, adding a head node to the linked list when the linked list is initialized, instead of using an extra typedef statement like in the book. At the same time, the L->data element of the head node stores the length of the linked list, thus eliminating the need to define Link-length. 2. For the merging function Merge of non-decreasing linked lists, please note that there is initialization and assignment of new linked lists inside the function. Every time one is added, space must be allocated for p->next. 3. The length of the linked list needs to be dynamically changed in the insertion and deletion functions. 4. The time complexity of the inversion function of the linked list is 0 (n). It should be noted that in order to make other functions usable, the inversion of the linked list does not simply change the position. It should be noted that the head node points to the first element and the next position of the tail element is NULL. That is to say, add p->next = r; L->next->next = NULL; L->next = p; statement. Note that when p->next==NULL, L->next still points to the first element of the previous linked list, and the element of the new linked list becomes the last one, so L->next->next==NULL must be added. 5. Regarding the difference between clearing and destroying linked lists.
