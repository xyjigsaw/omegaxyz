Basic operations of data structure tree (main traversal)

Programming LanguagesC & C++Data StructuresTechnologyC++Data Structures
one. Experiment purpose Implement the basic operations of trees in C language, mainly examining the pre-order, in-order, post-order and hierarchical traversal of trees. 2. Main content The binary tree used in this article is as follows: obviouslyPreface: ABCDEGF Mid-order: CBEGDFA Sequence: CGEFDBA Level:ABCDEFG BiTree.h content
typedef char TElemType;
typedef int Status; 
typedef struct BiTNode{
	TElemType data;
	struct BiTNode *lchild,*rchild;
}BiTNode,*BiTree;

Main functions: ① Create a binary tree in order Note that if there are no left or right subtrees when creating, you must enter spaces. Input: ABC_ _DE_G_ _F_ _ _
Status PreCreateBiTree(BiTree &T)
{
	char ch;
    ch=getchar();
	if(ch==' ')T=NULL;
	else
	{
		if(!(T=(BiTNode *)malloc(sizeof(BiTNode))))exit(OVERFLOW);
		T->data=ch;
		PreCreateBiTree(T->lchild);
		PreCreateBiTree(T->rchild);
	}
	return OK;
}
② Pre-order traversal (recursive algorithm)
Status PreOrderTraverse(BiTree T,Status(*Visit)(TElemType e))
{
	if(T){
		if(Visit(T->data))
			if(PreOrderTraverse(T->lchild,Visit))
				if(PreOrderTraverse(T->rchild,Visit))return OK;
		return ERROR;
	}else return OK;
}
③ In-order traversal (recursive algorithm)
Status InOrderTraverse2(BiTree T,Status(*Visit)(TElemType e))
{
	if(T){
		InOrderTraverse2(T->lchild,Visit);
		Visit(T->data);
		InOrderTraverse2(T->rchild,Visit);
	}
	return OK;
}
④ In-order traversal (non-recursive algorithm) Note that the C++STL header file include<stack> needs to be included here.
Status InOrderTraverse1(BiTree T,Status(*Visit)(TElemType e))
{
	stack<BiTree>S;
	BiTree p;
	S.push(T);
	while(!S.empty()){
		while(p=S.top())S.push(p->lchild);
		p=S.top();
		S.pop();
		if(!S.empty()){
			p=S.top();
		    S.pop();
			if(!Visit(p->data))return ERROR;
			S.push(p->rchild);
        }
		return OK;
	}
}
⑤ Post-order traversal (recursive algorithm)
Status PostOrderTraverse(BiTree T,Status(*Visit)(TElemType e))
{
	if(T){  
        PostOrderTraverse(T->lchild,Visit);
	    PostOrderTraverse(T->rchild,Visit);
		Visit(T->data);
	    }  

	return OK;
}
⑥Level traversal (using QUEUE) You can include STL<queue> or define an array, just use a circular queue.
Status LevelOrderTraverse(BiTree T,Status(*Visit)(TElemType e))
{
	BiTree p;     
	BiTNode *Q[100];     
	int front,rear;     
	front=rear=-1;    
	rear++;     
	Q[rear]=T;     
	while(front!=rear){         
		front=(front+1)%100;         
		p=Q[front];         
		Visit(p->data);         
		if(p->lchild!=NULL){             
			rear=(rear+1)%100;
			Q[rear]=p->lchild;
		}       
		if(p->rchild!=NULL){             
			rear=(rear+1)%100;             
			Q[rear]=p->rchild;         
		}          
	}    
	return OK;
}

Status Visit(TElemType e)

	printf("%c ",e);

⑧ Calculate the number of nodes in the tree
Status CountNode(BiTree T,int &d)
{
	if(T){
		d++;
		CountNode(T->lchild,d);
		CountNode(T->rchild,d);
	}
	return OK;
}
⑨ Calculate the depth of the tree
Status GetDepth(BiTree T)
{
	int hl,hr;
	if(T==NULL)return 0;
	else{
		hl=GetDepth(T->lchild);
		hr=GetDepth(T->rchild);
		if(hl>hr)return hl+1;
		else return hr+1;
	}
}

Experiment summary 1. The traversal function can be written as recursive or non-recursive, and the recursive function is more concise. 2. Hierarchical traversal requires the use of a queue, which can include C++STL<queue> or define an array and use a circular queue. Note that each time the judgment is made, the head of the queue must be assigned to the temporary variable P, and the left and right subtrees must be inserted from the end of the queue. 3. When creating a tree in order, please note that if there are no left and right subtrees, you must enter a space. Input: ABC_ _DE_G_ _F_ _ _
