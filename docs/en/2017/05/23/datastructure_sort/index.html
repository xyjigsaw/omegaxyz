typedef int KeyType;

typedef struct{

    RedType r[MAXSIZE+1];

①Insertion sort

void InsertSort(SqList &L)
{
    int i,j,a=0,b=0;
    for(i=1;i<=L.length;++i)
    {
        if(L.r[i].key<L.r[i-1].key)
        {
            L.r[0]=L.r[i];
            L.r[i]=L.r[i-1];
            a++;
        }
        for(j=i-2;L.r[0].key<L.r[j].key;--j)
            L.r[j+1]=L.r[j];b++;
        L.r[j+1]=L.r[0];
    }
    cout<<"比较次数:"<<a<<"移动次数:"<<b<<endl;
}

②Half insertion sort

void BInsertSort(SqList &L)
{
    int low,high,m;
    for(int i=2;i<=L.length;++i)
    {
        L.r[0]=L.r[i];
        low=1;high=i-1;
        while(low<=high)
        {
            m=(low+high)/2;
            if(L.r[0].key<L.r[m].key)high=m-1;
            else low=m+1;
        }
        for(int j=i-1;j>=high+1;--j)
            L.r[j+1]=L.r[j];
        L.r[high+1]=L.r[0];
    }
}

③Select sort

void SelectSort(SqList &L)
{
    int j,k;
    for(int i=1;i<=L.length;++i)
    {
        k=i;
        for(j=i+1;j<=L.length;j++)
            if(L.r[j].key<L.r[k].key)k=j;
        if(k!=i)
        {
            L.r[0].key=L.r[i].key;
            L.r[i].key=L.r[k].key;
            L.r[k].key=L.r[0].key;
        }
    }
}

④Bubble sorting

void BubbleSort(SqList &L)
{
    int i,j;
    for(i=1;i<=L.length;++i)
    {
        for(j=1;j<L.length-i+1;++j)
        {
            if(L.r[j+1].key<L.r[j].key)
            {
                L.r[0].key=L.r[j].key;
                L.r[j].key=L.r[j+1].key;
                L.r[j+1].key=L.r[0].key;
            }
        }
    }
}

⑤Quick sort

int Partition(SqList &L,int low,int high)
{
    L.r[0]=L.r[low];
    KeyType pivotkey=L.r[low].key;
    while(low<high)
    {
        while(low<high&&L.r[high].key>=pivotkey)--high;
        L.r[low]=L.r[high];
        while(low<high&&L.r[low].key<=pivotkey)++low;
        L.r[high]=L.r[low];
    }
    L.r[low]=L.r[0];
    return low;
}

void QSort(SqList &L,int low,int high)
{
    if(low<high)
    {
        int pivotloc=Partition(L,low,high);
        QSort(L,low,pivotloc-1);
        QSort(L,pivotloc+1,high);
    }
}

⑧Merge sort

void Merge(RedType SR[],RedType &TR[],int i,int m,int n)
{
    int j,k;
    for(j=m+1,k=i;i<=m&&j<=n;++k)
    {
        if(SR[i].key<=SR[j].key)
            TR[k]=SR[i++];
        else
            TR[k]=SR[j++];
    }
    int t;
    if(i<=m)
    {
        for(t=i;t<=m;t++)
            TR[k+t-i]=SR[t];
    }
    if(j<=n)
    {
        for(t=j;t<=m;t++)
            TR[k+t-j]=SR[t];
    }
}

void MSort(RedType SR[],RedType *TR1,int s,int t)
{
    int m;
    RedType TR2[MAXSIZE+1];
    if(s==t)TR1[s]=SR[s];
    else{
        m=(s+t)/2;
        MSort(SR,TR2,s,m);
        MSort(SR,TR2,m+1,t);
        Merge(TR2,TR1,s,m,t);
    }
}
void MergeSort(SqList &L)
{
    MSort(L.r,L.r,1,L.length);
}

4. Experiment summary (1) If n is small (for example, n<50), direct insertion sort, bubble sort or simple selection sort can be used. If there is a lot of data in the record and the movement is time-consuming, a simple selection sorting method should be used. (2) If the initial state of the records is basically ordered according to the key code, it is appropriate to use direct insertion sort or bubble sort. (3) If n is large, an improved sorting method should be used, such as quick sort, heap sort or merge sort. The time complexity of these sorting algorithms is O(nlog2n), but in terms of average performance, quick sort is considered to be the best sorting method in internal sorting based on comparing record keys. Unfortunately, the worst-case time complexity of quick sort is O(n2), and the worst-case time complexity of heap sort and merge sort is still O(nlog2n). Both heap sort and quick sort are unstable sorts. If stable sorting is required, merge sorting can be used. (4) Radix sorting can complete the sorting of n records in O (d×n) time. d refers to the number of single logical keys, which is generally much less than n. However, radix sorting is only suitable for keys with obvious structural characteristics such as strings and integers. (5) The sorting algorithms discussed previously, except for radix sorting, are all implemented on sequential storage. When the amount of information in the record itself is large, in order to avoid spending a lot of time moving data, a linked list can be used as a storage structure. Both insertion sort and merge sort are easy to implement on linked lists, but some sorting methods, such as quick sort and heap sort, are difficult to implement on linked lists.
