Data structure Huffman tree and encoding

Programming LanguagesC & C++Data StructuresTechnologyC++Data StructuresC
1. Experimental purpose Construct a Huffman tree, and find the Huffman tree code based on the constructed Huffman tree; 2. Basic ideas 1. Arrange each English letter from small to large according to the frequency of occurrence, with the smallest on the left to form a sequence 2. Each letter represents a terminal node (leaf node), compare the frequency of occurrence of each letter, add the minimum two letter frequencies to form a new node, delete the two letters from the sequence, and add the generated node to the letter queue 3. Repeat the previous two steps until there are no letters in the sequence To encode: 1. Give all left links '0' and right links '1' of the Huffman tree 2. Record the codes of all letters in order from the root of the tree to the leaves3. Basic code1.Huffman.h file
typedef struct{
	int weight;
	int parent,lchild,rchild;
}HTNode,*HuffmanTree;

typedef char **HuffmanCode;

void HuffmanCoding(HuffmanTree &HT,HuffmanCode &HC,int *w,int n);
void HuffmanSelect(HuffmanTree HT,int k,int &s1,int &s2);
2. Main functions ①select function
void HuffmanSelect(HuffmanTree HT,int n,int &s1,int &s2)
{  
	int minn=10000000,maxx=10000000;  
    s1=s2=0;  
	for(int i=1; i<=n; i++)  
	{  
	    if(HT[i].parent==0)  
	    {  
           if(HT[i].weight<minn)  
	       {  
               minn=HT[i].weight;s1=i;  	            }  
	       }  
	    }  
	    int t=HT[s1].weight;  
	    HT[s1].weight=maxx;  
	    minn=99999998;  
	    for(int i=1; i<=n; i++)  
	    {  
	        if(HT[i].parent==0)  
	        {  
	            if(HT[i].weight<minn)  
	            {  
	                minn=HT[i].weight;  
	                s2=i;  
	            }  
	        }  
	    }  
	    HT[s1].weight=t;  
}
②Encoding function
void HuffmanCoding(HuffmanTree &HT,HuffmanCode &HC,int *w,int n)
{
	if(n<=1)return;
	int m=2*n-1;
	HT=(HuffmanTree)malloc((m+1)*sizeof(HTNode));
	int i;
	for(i=1;i<=n;i++)
	{
		HT[i].weight=w[i];
		HT[i].parent=HT[i].rchild=HT[i].lchild=0;
	}
	for(;i<=m;i++)
	{
		HT[i].weight=HT[i].parent=HT[i].rchild=HT[i].lchild=0;
	}
	int s1=0,s2=0;
	for(i=n+1;i<=m;i++)
	{
		HuffmanSelect(HT,i-1,s1,s2);
		HT[s1].parent=i;HT[s2].parent=i;
		HT[i].lchild=s1;HT[i].rchild=s2;
		HT[i].weight=HT[s1].weight+HT[s2].weight;
	}
	HC=(HuffmanCode)malloc((n+1)*sizeof(char*));
	char *cd;
	cd=(char *)malloc(n*sizeof(char));
	cd[n-1]='\0';
	int start;
	for(i=1;i<=n;++i)
	{
		start=n-1;
		int c=0;int f=0;
		for(c=i,f=HT[i].parent;f!=0;c=f,f=HT[f].parent)
		{
			if(HT[f].lchild==c) 
				cd[--start]='0';
			else 
				cd[--start]='1';
		}
		HC[i]=(char*)malloc((n-start)*sizeof(char));
		strcpy(HC[i],&cd[start]);
	}
	free(cd);
}
