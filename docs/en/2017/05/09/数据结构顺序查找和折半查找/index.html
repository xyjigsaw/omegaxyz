Data structure sequential search and binary search

Programming LanguagesC & C++Data StructuresTechnologyC++Data Structures
Experiment purpose Use C language to implement sequential search and binary search in static lookup tables, and analyze the time.Search.hdocument 
#define OK 1
#define ERROR -1
#include <stdio.h>
#include<stdlib.h>
#include<time.h>
#define MAXSIZE 100
typedef int KeyType;
typedef int ElemType;

typedef struct{
	ElemType data[10];
	KeyType key;
}Node;

typedef struct{
	Node elem[MAXSIZE];
	int length;
}SSTable;

int CreateTable(SSTable &ST,int n);
void OutputTable(SSTable ST);
int Search_Seq(SSTable ST,KeyType key);
int Search_Bin(SSTable ST,KeyType key);
Main functions: ① Creation of sequence table
int CreateTable(SSTable &ST,int n)
{

	int i;
	ST.length=n;
	for(i=1;i<=n;i++)
	{
		printf("输入关键字：");
		scanf("%d",&ST.elem[i].key);
		printf("输入值：");
		scanf("%s",&ST.elem[i].data);
	}
	return OK;
}
②.Output of sequence table
void OutputTable(SSTable ST)
{
    int i;
    printf("关键字: ");
    for(i=1;i<=ST.length;i++)
        printf("%6d",ST.elem[i].key);
    printf("\n值    : ");
    for(i=1;i<=ST.length;i++)
        printf("  %s ",ST.elem[i].data);
}
③Sequential search
int Search_Seq(SSTable ST,KeyType key)
{
	ST.elem[0].key=key;
	int i=ST.length;
	while(ST.elem[i].key!=key)i--;
	return i;
}
④ Search by half
int Search_Bin(SSTable ST,KeyType key)
{
    int low=1,high=ST.length;
    while(low<=high)
    {
        int mid=(low+high)/2;
        if(key==ST.elem[mid].key)return mid;
        else if(key<ST.elem[mid].key)high=mid-1;
        else low=mid+1;
    }
    return 0;
}
Mainfunction  A time function is added to the Main function to test the search time. Of course, in the experiment, a large amount of data cannot be input, so the search time between the two is not much different.
int main(int argc, char* argv[])
{
	clock_t start,finish;
	printf("输入构造的顺序表的长度：");
	SSTable ST;int n;
	scanf("%d",&n);
	CreateTable(ST,n);
	printf("检查顺序表\n");
	OutputTable(ST);
	printf("\n--------------顺序查找--------------------\n");
	printf("输入要查找的值的序号：");
	int key;
    scanf("%d",&key);
    int t;
    start=clock();
    t=Search_Seq(ST,key);
    finish=clock();
    if(t==0)
        printf("查找失败，无匹配\n");
    else
        {
            printf("关键字为%d的数据是%s。\n",t,ST.elem[Search_Seq(ST,key)].data);
            printf("查找时间为: %lf\n",(double)(finish-start));
        }
    printf("\n--------------折半查找--------------------\n");
	printf("输入要查找的值的序号：");
	scanf("%d",&key);
	start=clock();
	t=Search_Bin(ST,key);
	finish=clock();
	if(t==0)
        printf("查找失败，无匹配\n");
    else
        {
            printf("关键字为%d的数据是%s。\n",t,ST.elem[Search_Bin(ST,key)].data);
            printf("查找时间为: %lf\n",(double)(finish-start));
        }
	return 0;
}
