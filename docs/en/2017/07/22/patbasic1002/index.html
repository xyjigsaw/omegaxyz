PAT Class B 1002 Write this number

Programming LanguagesC & C++TechnologyC++CPAT
Read in a natural number n, calculate the sum of its digits, and write each digit of the sum in Chinese Pinyin.Input format:Each test input contains 1 test case, which gives the value of the natural number n. Here it is guaranteed that n is less than 10100。 Output format:Output each digit of the sum of the digits of n in one line. There is 1 space between the Pinyin digits, but there is no space after the last Pinyin digit in the line.Input example: 1234567890987654321123456789 Output sample:yi san wu Code:
#include<iostream>
#include<cstdio>
#include<cstring>
using namespace std;

void print_n(int n,int a){
    switch(n){
        case 0:cout<<"ling";break;
        case 1:cout<<"yi";break;
        case 2:cout<<"er";break;
        case 3:cout<<"san";break;
        case 4:cout<<"si";break;
        case 5:cout<<"wu";break;
        case 6:cout<<"liu";break;
        case 7:cout<<"qi";break;
        case 8:cout<<"ba";break;
        case 9:cout<<"jiu";break;
        }
    if(a!=0)
        cout<<" ";
}

void pinyin(char *sum){
    int len=strlen(sum);
    int n=0,flag=0;
    for(int i=len-1;i>=0;i--){
        n=sum[i]-'0';
        print_n(n,i);
    }
}

int main(){
    char bigNum[101];
    while(scanf("%s",bigNum)!=EOF){
        int sum1=0,sum2=0,mod=0,flag=0;
        for(int i=0;i<strlen(bigNum);i++){
            sum1+=bigNum[i]-'0';
        }
        sum2=sum1;
        char sum_storage[5]={'\0'};
        while(sum2>0){
            mod=sum2%10;
            sum2/=10;
            sum_storage[flag]=mod+'0';
            flag++;
        }
        pinyin(sum_storage);
    }
}
