Python batch modify file names

Programming LanguagesTechnologyPythonPython
Use the find join rename method in python's os module to modify the file name with one click on file name prefixes such as [Movie Paradise www.dy2018.com] The main idea is to traverse the file and modify the strings that meet the conditions. The code is given below:
import os

def visitDir(path):
    if not os.path.isdir(path):
        print('Error: "', path, '" is not a directory or does not exist.')
        return
    else:
        global x

        for lists in os.listdir(path):
            try:
                sub_path = os.path.join(path, lists)
                x += 1
                if sub_path.find('[') > 0:
                    index1 = sub_path.index('[')
                    index2 = sub_path.index(']')
                    index3 = sub_path.index('.')
                    if sub_path[index2+1] == '.':
                        newname = sub_path[:index3-1] + sub_path[index3+1:len(sub_path)]
                    else:
                        newname = sub_path[:index1] + sub_path[index2 + 1:len(sub_path)]
                    os.rename(sub_path, newname)
                print('No.', x, ' ', newname, '修改成功！')
                if os.path.isdir(sub_path):
                    visitDir(sub_path)
            except:
                pass

if __name__ == '__main__':
    x = 0
    visitDir('H:\\Movie&Series')
