Python password strength judgment

Programming LanguagesTechnologyPythonPython
Determine whether a string meets strong password requirements without using regular expressions. (Length>=10, at least three of the following situations occur: uppercase letters, lowercase letters, numbers, symbol set (_!@#): and does not include other characters)
def judge(pw):
    if len(pw) < 10:
        return False
    flag = [0, 0, 0, 0]
    symbol = ['_', '!', '@', '#']
    for i in pw:
        if i.isupper():
            flag[0] = 1
        elif i.islower():
            flag[1] = 1
        elif i.isdigit():
            flag[2] = 1
        elif i in symbol:
            flag[3] = 1
        else:
            return False
    if sum(flag) >= 3:
        return True
    else:
        return False

if __name__ == '__main__':
    while True:
        if judge(input('input:')):
            print('OK')
            break
        else:
            print('ERROR')
