Python uses wxpython to make a simple text editor

Programming LanguagesTechnologyPythonPythonwxpythonGUI
The following python code implements a simple text editor that can implement editing, loading and saving operations. The entire interface is built using wxpython. wxpython is a GUI graphical interface library. Allows Python programmers to easily create a complete GUI user interface with full function keys. In addition, this text editor has some shortcomings, including using os.exists() to create new files, but the effect on already created files is OK. I will add various features to the next generation text editor. This text editor has no practical value, it is only for learning. The code is given below:
import wx
import os

app = wx.App()
win = wx.Frame(None, title="TXT Editor", size=(1000, 666))

bkg = wx.Panel(win)

def openFile(evt):
    dlg = wx.FileDialog(
        win,
        "Open",
        "",
        "",
        "All files (*.*)|*.*",
        wx.FD_OPEN | wx.FD_FILE_MUST_EXIST)
    filepath = ''
    if dlg.ShowModal() == wx.ID_OK:
        filepath = dlg.GetPath()
    else:
        return
    filename.SetValue(filepath)
    fopen = open(filepath)
    fcontent = fopen.read()
    contents.SetValue(fcontent)
    fopen.close()


def saveFile(evt):

    fcontent = contents.GetValue()
    fopen = open(filename.GetValue(), 'w')
    fopen.write(fcontent)
    fopen.close()
    win = wx.Frame(None, title='TXT Editor')
    button = wx.Button(win, label='保存成功')
    win.Show()

openBtn = wx.Button(bkg, label='浏览')
openBtn.Bind(wx.EVT_BUTTON, openFile)

saveBtn = wx.Button(bkg, label='保存')
saveBtn.Bind(wx.EVT_BUTTON, saveFile)

filename = wx.TextCtrl(bkg, style=wx.TE_READONLY)
contents = wx.TextCtrl(bkg, style=wx.TE_MULTILINE)

hbox = wx.BoxSizer()
hbox.Add(openBtn, proportion=0, flag=wx.LEFT | wx.ALL, border=5)
hbox.Add(filename, proportion=1, flag=wx.EXPAND | wx.TOP | wx.BOTTOM, border=5)
hbox.Add(saveBtn, proportion=0, flag=wx.LEFT | wx.ALL, border=5)

bbox = wx.BoxSizer(wx.VERTICAL)
bbox.Add(hbox, proportion=0, flag=wx.EXPAND | wx.ALL)
bbox.Add(contents, proportion=1, flag=wx.EXPAND | wx.LEFT | wx.BOTTOM | wx.RIGHT, border=5)

bkg.SetSizer(bbox)
win.Show()
app.MainLoop()
